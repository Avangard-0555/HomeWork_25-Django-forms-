# Установить PostgreSQL с официального сайта (https://www.postgresql.org/download/), если он еще не установлен.

# Создать базу данных в PostgreSQL
# Откройте терминал или командную строку и подключитесь к PostgreSQL:

psql -U postgres
#где postgres  это имя пользователя, если я не менял по умолчанию.


#Далее создайте базу данных:

CREATE DATABASE (имя моей бд)navi;   #Например, если сайт использует базу данных mysite, создай ее:   CREATE DATABASE mysite;


#Создайте таблицы для сайта
#Если вам нужно создать простую таблицу,
#например, для хранения пользователей, можно использовать 
#такую команду:

CREATE TABLE users (
  id SERIAL PRIMARY KEY,  -- ID пользователя, который будет уникальным
  name VARCHAR(16),      -- Имя пользователя, максимум 100 символов
  email VARCHAR(25) UNIQUE NOT NULL  -- Электронная почта, которая должна быть уникальной и не пустой
);


CREATE TABLE users — это создание таблицы с именем users.


id SERIAL PRIMARY KEY — создает поле id, которое 
будет уникальным для каждого пользователя и автоматически увеличиваться 
с каждым новым пользователем.


name VARCHAR(100) — поле для имени пользователя, 
где можно хранить до 100 символов.


email VARCHAR(100) UNIQUE NOT NULL — поле для почты,
где почта будет уникальной для каждого пользователя и не может быть

#Чтобы убедиться, что таблица создана,
#можно выполнить команду:

\dt

значит таблица создана успешно.
